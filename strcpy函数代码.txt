int length(char* p)

{
	int n = 0;
	while (*p != '\0')
	{
		n++;
		p++;
	}
	return n;

}

void copy(char* dest, char* src)
{
	int i, j = 0;
	for (i = 0; i < strlen(src); i++)
	{
		if (src[i] < '0' || src[i]>'9')
		{
			dest[j] = src[i];
			j++;
		}
	}
	printf("%s\n", dest);
}
